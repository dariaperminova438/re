# Python
def array_sum(arr, index=0):
    if index >= len(arr):
        return 0
    return arr[index] + array_sum(arr, index + 1)

# Тест 1: Обычный массив
numbers1 = [1, 3, 5, 7, 9]
result1 = array_sum(numbers1)
print(f"Тест 1 - Сумма {numbers1} = {result1}")

# Тест 2: Массив с отрицательными числами
numbers2 = [10, -2, 5, -8, 3]
result2 = array_sum(numbers2)
print(f"Тест 2 - Сумма {numbers2} = {result2}")

# Тест 3: Пустой массив
numbers3 = []
result3 = array_sum(numbers3)
print(f"Тест 3 - Сумма {numbers3} = {result3}")



# Java
  public class ArraySum {
    public static int arraySum(int[] arr, int index) {
        if (index >= arr.length) {
            return 0;
        }
        return arr[index] + arraySum(arr, index + 1);
    }
    
    public static void main(String[] args) {
        // Тест 1: Большой массив
        int[] numbers1 = {100, 200, 300, 400, 500};
        int result1 = arraySum(numbers1, 0);
        System.out.println("Тест 1 - Сумма больших чисел = " + result1);
        
        // Тест 2: Массив с нулями
        int[] numbers2 = {5, 0, 3, 0, 1};
        int result2 = arraySum(numbers2, 0);
        System.out.println("Тест 2 - Сумма с нулями = " + result2);
        
        // Тест 3: Один элемент
        int[] numbers3 = {42};
        int result3 = arraySum(numbers3, 0);
        System.out.println("Тест 3 - Сумма одного элемента = " + result3);
    }
}


# C++
#include <iostream>
#include <vector>

using namespace std;

int arraySum(const vector<int>& arr, int index) {
    if (index >= arr.size()) {
        return 0;
    }
    return arr[index] + arraySum(arr, index + 1);
}

int main() {
    // Тест 1: Дробные числа (целочисленный массив)
    vector<int> numbers1 = {15, 25, 35, 45};
    int result1 = arraySum(numbers1, 0);
    cout << "Тест 1 - Арифметическая прогрессия: " << result1 << endl;
    
    // Тест 2: Отрицательные и положительные
    vector<int> numbers2 = {-1, -2, -3, 10, 20};
    int result2 = arraySum(numbers2, 0);
    cout << "Тест 2 - Смешанные знаки: " << result2 << endl;
    
    // Тест 3: Все одинаковые числа
    vector<int> numbers3 = {7, 7, 7, 7, 7};
    int result3 = arraySum(numbers3, 0);
    cout << "Тест 3 - Все одинаковые: " << result3 << endl;
    
    return 0;
}
